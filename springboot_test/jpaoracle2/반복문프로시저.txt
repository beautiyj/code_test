-- boards 테이블에 동일한 데이터 300개를 넣는 프로시저
CREATE OR REPLACE PROCEDURE insert_boards_dummy
IS
BEGIN
    FOR i IN 1..300 LOOP
        INSERT INTO boards (no, writer, passwd, subject, content, regdate, updatedate)
        VALUES (boards_seq.NEXTVAL, '홍길동', '1234', 'JPA연습', 'JPA 내용', SYSTIMESTAMP, SYSTIMESTAMP);
    END LOOP;

    COMMIT;
END;
/

-- 프로시저 실행
EXEC insert_boards_dummy;
