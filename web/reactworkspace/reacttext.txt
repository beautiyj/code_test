# 1. PowerShell 권한 설정 (최초 1회만)
Set-ExecutionPolicy RemoteSigned

# 2. 프로젝트 생성 (둘 중 하나)
npx create-react-app 프로젝트명     # 구버전 방식
npm create vite@latest 프로젝트명   # 최신 방식 (권장)

# 3. 프로젝트 폴더 이동
cd 프로젝트명

# 4. 패키지 설치
npm install

# 5. 실행
npm run dev      # Vite
npm start        # create-react-app
